# 第十九章 负载均衡器安全

## 19.1 负载均衡概述

### 19.1.1 负载均衡的作用

负载均衡（Load Balancing）是分布式系统的核心组件，用于将传入的网络流量分发到多个服务器，提高系统的可用性、可扩展性和性能。

**负载均衡的主要功能**

1. **流量分发**：根据预设算法将请求分发到后端服务器
2. **健康检查**：定期检查后端服务器状态，自动剔除故障节点
3. **SSL终结**：在负载均衡器完成SSL/TLS解密，减轻后端压力
4. **会话保持**：确保同一用户的请求分发到同一后端服务器
5. **防御DDoS**：吸收和过滤恶意流量

### 19.1.2 常见负载均衡软件**

- **Nginx**：高性能HTTP反向代理和负载均衡
- **HAProxy**：TCP/HTTP负载均衡，支持L4/L7
- **LVS（Linux Virtual Server）**：Linux内核级负载均衡
- **Traefik**：云原生反向代理和负载均衡
- **Envoy**：C++开发的高性能服务代理

## 19.2 Nginx负载均衡安全配置

### 19.2.1 Nginx负载均衡基础配置

```nginx
# /etc/nginx/conf.d/upstream.conf

# 定义后端服务器池
upstream backend {
    # 最小连接数算法
    least_conn;
    
    # 后端服务器
    server 10.0.0.10:8080 weight=5 max_fails=3 fail_timeout=30s;
    server 10.0.0.11:8080 weight=5 max_fails=3 fail_timeout=30s;
    server 10.0.0.12:8080 weight=3 max_fails=3 fail_timeout=30s;
    
    # 备份服务器
    server 10.0.0.100:8080 backup;
    
    # 保持连接
    keepalive 32;
}

# HTTP负载均衡配置
server {
    listen 80;
    server_name example.com;
    
    location / {
        proxy_pass http://backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # 连接超时配置
        proxy_connect_timeout 30s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
        
        # 缓冲配置
        proxy_buffering on;
        proxy_buffer_size 4k;
        proxy_buffers 8 4k;
    }
}
```

### 19.2.2 Nginx HTTPS负载均衡配置

```nginx
# HTTPS负载均衡配置
server {
    listen 443 ssl http2;
    server_name example.com;
    
    # SSL证书配置
    ssl_certificate /etc/nginx/ssl/example.com.crt;
    ssl_certificate_key /etc/nginx/ssl/example.com.key;
    
    # SSL安全配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers 'ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256';
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    
    # 安全头部
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    
    # HSTS
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    
    location / {
        proxy_pass https://backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
        
        # SSL验证后端
        proxy_ssl_verify on;
        proxy_ssl_trusted_certificate /etc/nginx/ssl/ca.crt;
        proxy_ssl_server_name on;
    }
}
```

### 19.2.3 Nginx安全加固配置

```nginx
# 防止常见攻击
# 防止HTTP攻击
limit_req_zone $binary_remote_addr zone=req_limit:10m rate=10r/s;
limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

server {
    listen 80;
    server_name example.com;
    
    # 请求频率限制
    limit_req zone=req_limit burst=20 nodelay;
    limit_req_status 429;
    
    # 连接数限制
    limit_conn conn_limit 10;
    limit_conn_status 429;
    
    # 禁止目录遍历
    location / {
        if ($request_uri ~* \.\.) {
            return 403;
        }
    }
    
    # 防止SQL注入
    if ($query_string ~* "union.*select.*\() {
        return 403;
    }
    
    # 防止XSS
    if ($query_string ~* "<.*>") {
        return 403;
    }
    
    # 隐藏版本信息
    server_tokens off;
    proxy_hide_header X-Powered-By;
    
    # 防止点击劫持
    add_header X-Frame-Options "SAMEORIGIN" always;
    
    # 防止MIME类型嗅探
    add_header X-Content-Type-Options "nosniff" always;
}
```

## 19.3 HAProxy负载均衡安全配置

### 19.3.1 HAProxy基础配置

```ini
# /etc/haproxy/haproxy.cfg

# 全局配置
global
    log /dev/log local0
    log /dev/log local1 notice
    chroot /var/lib/haproxy
    stats socket /run/haproxy/admin.sock mode 660 level admin
    stats timeout 30s
    user haproxy
    group haproxy
    daemon
    
    # SSL配置
    ssl-default-bind-ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256
    ssl-default-bind-options no-sslv3 no-tlsv10 no-tlsv11
    
    # 性能优化
    maxconn 4096
    tune.ssl.default-dh-param 2048

# 默认配置
defaults
    log global
    mode http
    option httplog
    option dontlognull
    option http-server-close
    option forwardfor except 127.0.0.0/8
    option redispatch
    
    # 超时配置
    timeout connect 5000ms
    timeout client 50000ms
    timeout server 50000ms
    
    # 速率限制
    timeout client-fin 30s
    timeout http-request 10s
    timeout http-keep-alive 10s
    
    # 安全选项
    http-check expect status 200
    capture request header Host len 64
    capture request header X-Forwarded-For len 100

# 统计页面配置
listen stats
    bind *:8404
    stats enable
    stats uri /stats
    stats refresh 30s
    stats admin if LOCALHOST

# 后端服务器定义
backend web_backend
    mode http
    balance roundrobin
    
    # 健康检查
    option httpchk GET /health.html
    http-check expect status 200
    
    # 后端服务器
    server web1 10.0.0.10:8080 check inter 2000 rise 2 fall 3 weight 100
    server web2 10.0.0.11:8080 check inter 2000 rise 2 fall 3 weight 100
    server web3 10.0.0.12:8080 check inter 2000 rise 2 fall 3 weight 80 backup
    
    # SSL配置
    ssl verify required
    ca-file /etc/haproxy/ssl/ca.crt
    
    # 启用XFF头
    option forwardfor header X-Real-IP

# 前端配置
frontend http_front
    bind *:80
    bind *:443 ssl crt /etc/haproxy/ssl/example.com.pem
    
    # 默认后端
    default_backend web_backend
    
    # ACL规则
    acl is_api path_beg /api/
    acl is_admin path_beg /admin/
    acl is_static path_end .jpg .png .css .js
    
    # 访问控制
    block if is_admin !src 10.0.0.0/8
    block if { src bad_actors }
    
    # 重定向规则
    http-request redirect scheme https unless { ssl_fc }
```

### 19.3.2 HAProxy安全加固

```ini
# 安全相关的ACL定义
# 编辑haproxy.cfg

# 定义IP黑名单
acl blacklist src -f /etc/haproxy/blacklist.txt

# 定义速率限制
acl exceeds_rate src_http_req_rate(hdr_ip) gt 100

# 定义异常请求
acl suspect_request url_reg \.\./ || url_reg \.sh$ || url_reg \.cgi$

# 应用ACL
tcp-request content reject if blacklist
http-request deny if exceeds_rate
http-request deny if suspect_request

# 启用日志审计
# 添加到defaults
log-format "%ci:%cp [%tr] %ft %b/%s %TR/%Tw/%Tc/%Tr/%Ta %ST %B %CC %CS %tsc %ac/%fc/%bc/%sc/%rc %sq/%bq %hr %hs %{+Q}r"

# 生成黑名单文件
echo "192.0.2.100" > /etc/haproxy/blacklist.txt
echo "198.51.100.50" >> /etc/haproxy/blacklist.txt
```

### 19.3.3 HAProxy高可用配置

```ini
# Keepalived配置实现HAProxy高可用

# /etc/keepalived/keepalived.conf

global_defs {
    router_id haproxy_master
    script_user root
    enable_script_security
}

vrrp_script haproxy_check {
    script "/usr/local/bin/haproxy_check.sh"
    interval 2
    weight 2
    fall 2
    rise 1
}

vrrp_instance VI_1 {
    state MASTER
    interface eth0
    virtual_router_id 51
    priority 100
    advert_int 1
    
    authentication {
        auth_type PASS
        auth_pass mysecretpass
    }
    
    virtual_ipaddress {
        10.0.0.100/24 dev eth0
    }
    
    track_script {
        haproxy_check
    }
}

# 健康检查脚本
# /usr/local/bin/haproxy_check.sh
#!/bin/bash
if [ $(ps -C haproxy --no-headers | wc -l) -eq 0 ]; then
    exit 1
fi
exit 0
```

## 19.4 LVS负载均衡配置

### 19.4.1 LVS NAT模式配置

```bash
# 启用IP转发
echo 1 > /proc/sys/net/ipv4/ip_forward

# 加载ip_vs模块
modprobe ip_vs
modprobe ip_vs_wrr  # 加权轮询调度

# 配置LVS NAT
ipvsadm -A -t 10.0.0.100:80 -s wrr
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.10:80 -m -w 5
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.11:80 -m -w 3
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.12:80 -m -w 2

# 查看配置
ipvsadm -L -n

# 保存配置
ipvsadm-save > /etc/sysconfig/ipvsadm

# 启动服务
systemctl enable ipvsadm
systemctl start ipvsadm
```

### 19.4.2 LVS DR模式配置

```bash
# Director服务器配置
# 加载模块
modprobe ip_vs
modprobe ip_vs_wrr

# 配置VIP
ip addr add 10.0.0.100/32 dev eth0 label eth0:0

# 抑制ARP响应
echo 1 > /proc/sys/net/ipv4/conf/all/arp_ignore
echo 2 > /proc/sys/net/ipv4/conf/all/arp_announce
echo 1 > /proc/sys/net/ipv4/conf/eth0/arp_ignore
echo 2 > /proc/sys/net/ipv4/conf/eth0/arp_announce

# 配置LVS
ipvsadm -C
ipvsadm -A -t 10.0.0.100:80 -s wrr
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.10:80 -g -w 5
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.11:80 -g -w 3
ipvsadm -a -t 10.0.0.100:80 -r 10.0.1.12:80 -g -w 2

# RealServer配置（每台后端服务器）
# 配置VIP
ip addr add 10.0.0.100/32 dev lo label lo:0

# 抑制ARP响应
cat >> /etc/sysctl.conf << 'EOF'
net.ipv4.conf.all.arp_ignore = 1
net.ipv4.conf.all.arp_announce = 2
net.ipv4.conf.lo.arp_ignore = 1
net.ipv4.conf.lo.arp_announce = 2
EOF

sysctl -p
```

## 19.6 实战案例：Web服务负载均衡高可用部署

### 19.6.1 案例需求

- 3台Web服务器
- 负载均衡器实现高可用
- HTTPS终结
- 自动健康检查
- 会话保持
- DDoS防护

### 19.6.2 部署步骤

**第一步：准备环境**

```bash
# 服务器规划
# LVS Director: 10.0.0.10 (MASTER), 10.0.0.11 (BACKUP)
# VIP: 10.0.0.100
# RealServer: 10.0.1.10, 10.0.1.11, 10.0.1.12

# 安装依赖
yum install -y keepalived ipvsadm haproxy nginx
```

**第二步：配置HAProxy**

```bash
# /etc/haproxy/haproxy.cfg
cat > /etc/haproxy/haproxy.cfg << 'EOF'
global
    log /dev/log local0
    chroot /var/lib/haproxy
    stats socket /run/haproxy/admin.sock mode 660 level admin
    user haproxy
    group haproxy
    daemon
    maxconn 4096
    ssl-default-bind-ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256

defaults
    log global
    mode http
    option httplog
    option dontlognull
    option http-server-close
    option forwardfor
    timeout connect 5000ms
    timeout client 50000ms
    timeout server 50000ms

# HTTP到HTTPS重定向
frontend http_front
    bind *:80
    default_backend http_back

backend http_back
    redirect scheme https code 301 if !{ ssl_fc }

# HTTPS前端
frontend https_front
    bind *:443 ssl crt /etc/ssl/private/server.pem
    default_backend web_backend
    
    # ACL规则
    acl static path_end .jpg .png .css .js .ico .svg .woff .woff2
    acl api path_beg /api/
    
    # 静态资源直接返回
    use_backend static_backend if static
    
    # API限流
    stick-table type ip size 100k expire 30s
    http-request track-sc0 src
    http-request deny deny_status 429 if { src_get_gpc0(http_front) gt 100 }

backend web_backend
    balance roundrobin
    option httpchk GET /health.html
    http-check expect status 200
    
    server web1 10.0.1.10:8080 check inter 2000 fall 3 rise 2
    server web2 10.0.1.11:8080 check inter 2000 fall 3 rise 2
    server web3 10.0.1.12:8080 check inter 2000 fall 3 rise 2 backup
    
    # 会话保持
    cookie SERVERID insert indirect nocache

backend static_backend
    balance roundrobin
    server static1 10.0.1.20:80 check
    server static2 10.0.1.21:80 check backup
EOF

systemctl enable haproxy
systemctl start haproxy
```

**第三步：配置Keepalived高可用**

```bash
# Master配置 /etc/keepalived/keepalived.conf
cat > /etc/keepalived/keepalived.conf << 'EOF'
global_defs {
    router_id haproxy_master
}

vrrp_script haproxy_check {
    script "/usr/local/bin/health_check.sh"
    interval 2
    weight 2
}

vrrp_instance VI_1 {
    state MASTER
    interface eth0
    virtual_router_id 51
    priority 100
    advert_int 1
    
    authentication {
        auth_type PASS
        auth_pass mysecretpass
    }
    
    virtual_ipaddress {
        10.0.0.100/24
    }
    
    track_script {
        haproxy_check
    }
}
EOF

# Backup配置相同，修改state为BACKUP，priority为90

# 健康检查脚本
cat > /usr/local/bin/health_check.sh << 'EOF'
#!/bin/bash
if [ $(ps -C haproxy --no-headers | wc -l) -eq 0 ]; then
    exit 1
fi
exit 0
EOF

chmod +x /usr/local/bin/health_check.sh

systemctl enable keepalived
systemctl start keepalived
```

